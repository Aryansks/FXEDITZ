    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // DOM Elements
            const loginSection = document.getElementById('loginSection');
            const mainContent = document.getElementById('mainContent');
            const accessCodeInput = document.getElementById('accessCode');
            const submitCodeBtn = document.getElementById('submitCode');
            const bookingModal = document.getElementById('bookingModal');
            const subscriptionModal = document.getElementById('subscriptionModal');
            const closeModalBtns = document.querySelectorAll('.close-modal');
            const bookBtns = document.querySelectorAll('.book-btn');
            const subscribeBtns = document.querySelectorAll('.subscribe-btn');
            const confirmBookingBtn = document.getElementById('confirmBooking');
            const confirmSubscriptionBtn = document.getElementById('confirmSubscription');
            const notification = document.getElementById('notification');
            const notificationText = document.getElementById('notificationText');
            const subscriptionStatusSection = document.getElementById('subscriptionStatus');
            const statusContent = document.getElementById('statusContent');
            
            // Codes
            const NORMAL_CODE = "FXANSH";
            const DEV_CODE = "FXANSHPRO";
            const CURRENT_COUPON = "ANSHKING";
            
            // Check if user has already entered code
            if (localStorage.getItem('userCode')) {
                loginSection.classList.add('hidden');
                mainContent.classList.remove('hidden');
            }
            
            // Submit code handler
            submitCodeBtn.addEventListener('click', function() {
                const code = accessCodeInput.value.trim().toUpperCase();
                
                if (code === NORMAL_CODE || code === DEV_CODE) {
                    // Save to localStorage
                    localStorage.setItem('userCode', code);
                    
                    // Show main content
                    loginSection.classList.add('hidden');
                    mainContent.classList.remove('hidden');
                    mainContent.classList.add('fadeIn');
                    
                    // If developer, show additional options
                    if (code === DEV_CODE) {
                        showNotification("Developer access granted. You can view all bookings.");
                    }
                } else {
                    showNotification("Invalid code. Please try again.", "error");
                    accessCodeInput.classList.add('shake');
                    setTimeout(() => {
                        accessCodeInput.classList.remove('shake');
                    }, 500);
                }
            });
            
            // Book button handlers
            bookBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    const service = this.getAttribute('data-service');
                    const price = this.getAttribute('data-price');
                    
                    document.getElementById('serviceName').value = service;
                    document.getElementById('servicePrice').value = `₹${price}`;
                    
                    bookingModal.classList.add('active');
                });
            });
            
            // Subscribe button handlers
            subscribeBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    const plan = this.getAttribute('data-plan');
                    const price = this.getAttribute('data-price');
                    const duration = this.getAttribute('data-duration');
                    
                    document.getElementById('subscriptionPlan').value = plan;
                    document.getElementById('subscriptionPrice').value = `₹${price}`;
                    
                    subscriptionModal.classList.add('active');
                });
            });
            
            // Close modal handlers
            closeModalBtns.forEach(btn => {
                btn.addEventListener('click', function() {
                    bookingModal.classList.remove('active');
                    subscriptionModal.classList.remove('active');
                });
            });
            
            // Confirm booking handler
            confirmBookingBtn.addEventListener('click', function() {
                const service = document.getElementById('serviceName').value;
                const price = document.getElementById('servicePrice').value;
                const name = document.getElementById('customerName').value;
                const email = document.getElementById('customerEmail').value;
                const phone = document.getElementById('customerPhone').value;
                const coupon = document.getElementById('couponCode').value;
                
                if (!name || !email || !phone) {
                    showNotification("Please fill all fields", "error");
                    return;
                }
                
                // Check coupon
                let finalPrice = price;
                if (coupon && coupon.toUpperCase() === CURRENT_COUPON) {
                    finalPrice = `₹${parseInt(price.replace('₹', '')) * 0.9}`; // 10% discount
                    showNotification(`Coupon applied! ${price} -> ${finalPrice}`);
                }
                
                // In a real app, you would send this data to your backend
                // For now, we'll just show a notification
                
                // Send message to developer
                const userCode = localStorage.getItem('userCode');
                if (userCode === DEV_CODE) {
                    // Don't send message if it's the developer booking
                    showNotification("Booking confirmed! (Developer booking)");
                } else {
                    // This would be sent to the developer in a real app
                    console.log(`New booking from ${name} (${email}): ${service} for ${finalPrice}`);
                    showNotification("Booking confirmed! Message sent to developer.");
                }
                
                // Close modal
                bookingModal.classList.remove('active');
                
                // Clear form
                document.getElementById('customerName').value = '';
                document.getElementById('customerEmail').value = '';
                document.getElementById('customerPhone').value = '';
                document.getElementById('couponCode').value = '';
            });
            
            // Confirm subscription handler
            confirmSubscriptionBtn.addEventListener('click', function() {
                const plan = document.getElementById('subscriptionPlan').value;
                const price = document.getElementById('subscriptionPrice').value;
                const name = document.getElementById('subCustomerName').value;
                const email = document.getElementById('subCustomerEmail').value;
                
                if (!name || !email) {
                    showNotification("Please fill all fields", "error");
                    return;
                }
                
                // In a real app, you would process payment here
                // For demonstration, we'll simulate a successful subscription
                
                // Calculate expiry date
                const durationDays = parseInt(document.querySelector('.subscribe-btn[data-plan="' + plan + '"]').getAttribute('data-duration'));
                const expiryDate = new Date();
                expiryDate.setDate(expiryDate.getDate() + durationDays);
                
                // Save to localStorage
                localStorage.setItem('userSubscription', JSON.stringify({
                    plan: plan,
                    price: price,
                    expiry: expiryDate.toISOString(),
                    status: 'pending' // Will be approved by developer
                }));
                
                // Send message to developer
                const userCode = localStorage.getItem('userCode');
                if (userCode === DEV_CODE) {
                    // Auto-approve if developer
                    approveSubscription();
                    showNotification("Subscription purchased and approved! (Developer)");
                } else {
                    // Notify developer for approval
                    console.log(`New subscription from ${name} (${email}): ${plan} for ${price}`);
                    showNotification("Subscription requested! Waiting for developer approval.");
                }
                
                // Close modal
                subscriptionModal.classList.remove('active');
                
                // Clear form
                document.getElementById('subCustomerName').value = '';
                document.getElementById('subCustomerEmail').value = '';
                
                // Show subscription status
                showSubscriptionStatus();
            });
            
            // Show subscription status
            function showSubscriptionStatus() {
                const subscription = JSON.parse(localStorage.getItem('userSubscription') || 'null');
                
                if (subscription) {
                    subscriptionStatusSection.classList.remove('hidden');
                    
                    const expiryDate = new Date(subscription.expiry);
                    const now = new Date();
                    const daysRemaining = Math.ceil((expiryDate - now) / (1000 * 60 * 60 * 24));
                    
                    let statusHTML = `
                        <div style="background: #e8f5e8; padding: 20px; border-radius: 10px;">
                            <h3>${subscription.plan} Plan</h3>
                            <p><strong>Status:</strong> <span style="color: ${subscription.status === 'approved' ? 'green' : 'orange'}">${subscription.status}</span></p>
                    `;
                    
                    if (subscription.status === 'approved') {
                        statusHTML += `
                            <p><strong>Expiry Date:</strong> ${expiryDate.toLocaleDateString()}</p>
                            <p><strong>Days Remaining:</strong> ${daysRemaining}</p>
                        `;
                    } else {
                        statusHTML += `<p>Your subscription is pending approval from the developer.</p>`;
                    }
                    
                    statusHTML += `</div>`;
                    
                    statusContent.innerHTML = statusHTML;
                }
            }
            
            // Approve subscription (simulate developer action)
            function approveSubscription() {
                const subscription = JSON.parse(localStorage.getItem('userSubscription') || 'null');
                if (subscription) {
                    subscription.status = 'approved';
                    localStorage.setItem('userSubscription', JSON.stringify(subscription));
                    showSubscriptionStatus();
                }
            }
            
            // Show notification
            function showNotification(message, type = "success") {
                notificationText.textContent = message;
                
                if (type === "error") {
                    notification.style.background = "#ff6b6b";
                } else {
                    notification.style.background = "";
                }
                
                notification.classList.add('active');
                
                setTimeout(() => {
                    notification.classList.remove('active');
                }, 3000);
            }
            
            // Initialize subscription status if exists
            showSubscriptionStatus();
            
            // Allow Enter key to submit code
            accessCodeInput.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    submitCodeBtn.click();
                }
            });
